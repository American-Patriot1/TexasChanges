### Checks if THIS is mutually neutral with PREV
is_neutral_country = {
	custom_trigger_tooltip = {
		NOT = {
			has_war_with = PREV
			is_ally_with = PREV
			# Combined into is_ally_with. Keeping this commented out for reference @Tran
			# is_subject_of = PREV
			# is_in_faction_with = PREV
			has_military_access_to = PREV
			gives_military_access_to = PREV
			is_justifying_wargoal_against = PREV
			has_wargoal_against = PREV
			PREV = {
				OR = {
					is_justifying_wargoal_against = PREV
					is_guaranteed_by = PREV
					# Unnecessary due to above is_ally_with covering this check both ways. Keeping this commented out for reference @Tran
					# is_subject_of = PREV
					has_wargoal_against = PREV
				}
			}
		}
		tooltip = is_neutral_country_tt
	}
}
